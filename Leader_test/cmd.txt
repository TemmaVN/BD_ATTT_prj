# Quét toàn bộ mạng để tìm Windows 10
sudo nmap -sn 192.168.1.0/24

# Quét với nhiều phương pháp
sudo nmap -sn -PE -PS21,22,23,25,80,135,139,445 192.168.1.0/24

# Lưu kết quả ra file
sudo nmap -sn 192.168.1.0/24 -oN host_discovery.txt
# Quét cổng cơ bản

# Quét tất cả cổng TCP
sudo nmap -p- [IP_Windows10]

# Quét cổng phổ biến Windows

sudo nmap -sU -p 53,67,68,69,123,137,138,161,445 [IP_Windows10]
# Xác định dịch vụ và OS
sudo nmap -A [IP_Windows10]

# Quét cổng với timing nhanh
sudo nmap -T4 -F [IP_Windows10]

# Quét kỹ các dịch vụ Windows
sudo nmap -sV -sC -p 135,139,445 [IP_Windows10]

# Phân tích với các tool khác
sudo netdiscover -r 192.168.1.0/24

# Tìm Windows host nhanh
sudo nmap -T4 -p 135,139,445 --open 192.168.1.0/24

# Quét SMB chi tiết
sudo nmap --script smb-os-discovery,smb-security-mode,smb-enum-shares [IP_Windows]

# Kiểm tra vulnerability
sudo nmap --script smb-vuln* [IP_Windows]
